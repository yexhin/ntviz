import seaborn as sns
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# solution plan
# i. The most suitable chart type is a bar chart because it effectively displays the average yield for each site. Error bars can be added to represent the standard deviation, providing a measure of variability.
# ii. Data transformation involves grouping the data by 'site' and calculating the mean and standard deviation of 'yield' for each site.
# iii.  Since we are comparing average yield across different sites, a simple bar chart is sufficient. No grouped or stacked bar chart or facet grid is needed.
# iv. Aesthetics will include clear axis labels, a title explaining the visualization, and appropriately colored bars with error bars.  A legend is not strictly necessary as the site names are displayed on the x-axis.


def plot(data: pd.DataFrame):
    # Calculate the mean and standard deviation of yield for each site
    site_stats = data.groupby('site')['yield'].agg(['mean', 'std'])

    # Create the bar plot
    plt.figure(figsize=(10, 6))  # Adjust figure size if needed
    ax = sns.barplot(x=site_stats.index, y=site_stats['mean'], yerr=site_stats['std'], capsize=0.1)

    # Customize the plot
    plt.xlabel('Site', fontsize=12)
    plt.ylabel('Average Yield', fontsize=12)
    plt.xticks(rotation=45, ha='right') # Rotate x-axis labels for better readability
    plt.title("Is there a significant difference in average yield (`yield`) between different crop sites (`site`)?", wrap=True)
    plt.tight_layout() # Adjust layout to prevent labels from overlapping

    return plt

chart = plot(data)
